window.startGoogleLoginDirect = async function startGoogleLoginDirect() { const message = document.getElementById('authMessage'); const button = document.getElementById('googleLoginButton'); const setMessage = (text, type = '') => { if (!message) return; message.textContent = text; message.className = `auth-message ${type}`; }; try { setMessage('Google 로그인으로 이동하고 있어요.', ''); if (!window.supabase?.createClient) { setMessage('Supabase 로딩 지연. 새로고침 후 다시 시도해 주세요.', 'error'); return; } const url = window.SCHEDULE_SUPABASE_URL; const key = window.SCHEDULE_SUPABASE_PUBLISHABLE_KEY; const client = window.__scheduleSupabaseClient || window.supabase.createClient(url, key, { auth: { persistSession: true, autoRefreshToken: true, detectSessionInUrl: false, // 자동 감지 끄기 (수동 제어) flowType: 'implicit' // PKCE 대신 implicit 사용 } }); window.__scheduleSupabaseClient = client; if (button) { button.disabled = true; button.textContent = 'Google 로그인으로 이동 중...'; } const redirectTo = `${window.location.origin}${window.location.pathname}`; const { error } = await client.auth.signInWithOAuth({ provider: 'google', options: { redirectTo } }); if (error) { setMessage(`로그인 시작 실패: ${error.message}`, 'error'); if (button) { button.disabled = false; button.textContent = 'Google로 로그인'; } } } catch (error) { setMessage(`처리 오류: ${error?.message || error}`, 'error'); if (button) { button.disabled = false; button.textContent = 'Google로 로그인'; } } };
